 * @class client.ui.Sport
 * Item to render a sport.

    displayName: 'sport',
 
 
 
    contextTypes: {
        data : React.PropTypes.object,
        lang : React.PropTypes.object
    },
 
 
 
    render: function(){
        return React.DOM.a({
            className : 'sport item',
            href      : '/' + this.context.data.lang + '/sports/' + this.props.id

    }
